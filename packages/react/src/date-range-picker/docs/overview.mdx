```jsx live
() => {
	// Set the value to a value that the user might think is valid
	const [value, setValue] = React.useState({
		from: undefined,
		to: undefined,
	});

	// This logic is for documentation purposes only. This should be done with `yup` or `zod`.
	// See Single-page form template
	const isFromInvalid = React.useCallback((value, otherDate) => {
		return value ? !isValidDate(value, { toDate: otherDate }) : false;
	}, []);

	const isToInvalid = React.useCallback((value, otherDate) => {
		return value ? !isValidDate(value, { fromDate: otherDate }) : false;
	}, []);

	const fromInvalid = isFromInvalid(value.from, value.to);
	const toInvalid = isToInvalid(value.to, value.from);

	const onChange = (dateRange) => {
		console.log('onChange', dateRange);
		setValue(dateRange);
	};

	return (
		<DateRangePicker
			value={value}
			onChange={onChange}
			fromInvalid={fromInvalid}
			toInvalid={toInvalid}
			message={
				fromInvalid && toInvalid
					? 'Enter valid start and end dates'
					: fromInvalid
					? 'Enter a valid start date'
					: toInvalid
					? 'Enter a valid end date'
					: undefined
			}
		/>
	);
};
```

```jsx live
() => {
	// Set the value to a value that the user might think is valid
	const [value, setValue] = React.useState({
		from: undefined,
		to: undefined,
	});

	const today = new Date();
	const lastWeek = new Date(
		today.getFullYear(),
		today.getMonth(),
		today.getDate() - 7
	);
	const nextWeek = new Date(
		today.getFullYear(),
		today.getMonth(),
		today.getDate() + 7
	);

	// This logic is for documentation purposes only. This should be done with `yup` or `zod`.
	// See Single-page form template

	const isInvalid = React.useCallback((value) => {
		return value
			? !isValidDate(value, { maxDate: nextWeek, minDate: lastWeek })
			: false;
	}, []);

	const fromInvalid = isInvalid(value.from);
	const toInvalid = isInvalid(value.to);

	const onChange = (dateRange) => {
		console.log('onChange', dateRange);
		setValue(dateRange);
	};

	return (
		<DateRangePicker
			value={value}
			onChange={onChange}
			fromInvalid={fromInvalid}
			toInvalid={toInvalid}
			minDate={lastWeek}
			maxDate={nextWeek}
			message={
				fromInvalid && toInvalid
					? 'Enter valid start and end dates'
					: fromInvalid
					? 'Enter a valid start date'
					: toInvalid
					? 'Enter a valid end date'
					: undefined
			}
		/>
	);
};
```
